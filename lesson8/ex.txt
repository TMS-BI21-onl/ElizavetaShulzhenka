INSERT INTO MillionPeople (FirstName, LastName)
SELECT TOP 1000000
     n1.first_name,
     n2.last_name
FROM [dbo].[FIRST_NAMES] n1
JOIN [dbo].[LAST_NAMES] n2;

 SELECT COUNT(FirstName)
