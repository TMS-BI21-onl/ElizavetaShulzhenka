EX3.

SELECT name, price
FROM product
WHERE price = (SELECT MAX(price) FROM product)

SELECT TOP 1 WITH TIES price, name
FROM product
order by price DESC

SELECT *
FROM
(SELECT name, price, 
		RANK() OVER (ORDER BY price DESC) as rnk
FROM product) T
WHERE rnk=1
